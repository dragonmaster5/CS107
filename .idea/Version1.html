    public void encrypt()
    { String[] plainTextToEncrypt = splitText(plainText);

    for (String s : plainTextToEncrypt) {
    Matrix Encrypted = Matrix.multiplyMatrix(keyMatrix, new Matrix(s, 1));
    for (int i = 0; i < 3; i++) {
    cipherText += (char) (Encrypted.getMatrix()[i][0] + 'A');
    }
    }

    }
    public void decrypt()
    {String[] CipherTextToDecrypt = splitText(cipherText);

    for (String s : CipherTextToDecrypt) {
    Matrix inverse = keyMatrix.Inverse();
    Matrix Encrypted = Matrix.multiplyMatrix(inverse, new Matrix(s, 1));
    for (int i = 0; i < 3; i++) {
    plainText += (char) (Encrypted.getMatrix()[i][0] + 'A');
    }
    }

    }


    public static String[] splitText(String ciphertext) {
    String[] subtexts = new String[ciphertext.length()/3];
    Arrays.fill(subtexts, "");
    int c=0;

    for (int i = 0; i < ciphertext.length(); i+=3) {
    if(c>subtexts.length)
    break;
    subtexts[c++] += ciphertext.substring(i,i+3);
    }

    return subtexts;


    }
